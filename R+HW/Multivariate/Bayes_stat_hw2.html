triangle_sample <- function(){
  while (TRUE) {
    x <- runif(1, 0, 2) #  
    u <- runif(1, 0, 1) #    
    a <- ifelse(x <= 1, x, 2-x) #  
    if (u <= a) {
      return(x)
    }
  }
}
